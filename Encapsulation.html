                class mobil {
                    private function jalan()
                    {
                        echo 'Mobil Berjalan';
                    }

                    protected function roda()
                    {
                        echo 'Roda ada 4';
                    }

                    public function jumlahroda()
                    {
                        echo $this->roda();
                    }

                    public function aktifkanjalan()
                    {
                        echo $this->jalan();
                    }
                }
                $Toyota = new mobil();
                echo $Toyota->jumlahroda();
                echo $Toyota->aktifkanjalan();
