Kode : 

			

		class Buku {
			public $judul;
			public $penulis;

			public function __construct($judul, $penulis) {
				$this->judul = $judul;
				$this->penulis = $penulis;
			}

			public function deskripsi() {
				return "Buku '{$this->judul}' ditulis oleh {$this->penulis}.";
			}
		}

		$buku1 = new Buku("Python untuk Pemula", "John Doe");
		$buku2 = new Buku("Mastering Machine Learning", "Jane Smith");

		echo $buku1->deskripsi();
		echo $buku2->deskripsi();

Hasil :

				

Buku 'Python untuk Pemula' ditulis oleh John Doe.
Buku 'Mastering Machine Learning' ditulis oleh Jane Smith.
